const SHEET_NAME = 'Responses'; const HEADERS = [ 'respondent_id', 'timestamp', 'duration_seconds', 'duration_text', 'block', 'gender', 'study_year', 'residence', 'monthly_expense', 'daily_distance', 'main_transport', 'ev_experience', 'purchase_intent', 'bank_name', 'bank_account', 'bank_holder', 'set1','set2','set3','set4','set5','set6','set7','set8', 'set9','set10','set11','set12','set13','set14','set15','set16', ]; function doPost(e) { try { const data = JSON.parse(e.postData.contents); appendRow(data); return ContentService .createTextOutput(JSON.stringify({ status: 'ok' })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: 'error', message: err.message })) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({ status: 'E2W Survey endpoint is live (16 sets, 2 blocks, timer + bank)' })) .setMimeType(ContentService.MimeType.JSON); } function appendRow(data) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(HEADERS); const headerRange = sheet.getRange(1, 1, 1, HEADERS.length); headerRange.setFontWeight('bold'); headerRange.setBackground('#1a4e8a'); headerRange.setFontColor('#ffffff'); sheet.setFrozenRows(1); } if (sheet.getLastRow() === 0) { sheet.appendRow(HEADERS); } const row = HEADERS.map(h => (data[h] !== undefined ? data[h] : '')); sheet.appendRow(row); }
